Typography

檔案名稱：sass/_typography.scss

字型基本設定

                
// 字型設定
$font-family:               'Lato,"PingFang TC","Helvetica Neue",Helvetica,微軟正黑體,新細明體,Arial,sans-serif';
$font-size:                 16px;

Verticla Rhythm設定

                
$base-font-size: 16px;              //字級16px
$base-line-height: 16px;            //行高16px
$rhythm-unit: 'em';                 //輸出單位：em
$base-font-stack:$font-family;      //於變數設定font family

//以下只有在CP區塊文章才會設定
.cp {
    h1 {
        @include adjust-font-size-to($base-font-size*2, 3);
        @include leader(.5);
        @include trailer(.5);
    }
    h2 {
        @include adjust-font-size-to($base-font-size*1.5, 3);
        @include leader(.5);
        @include trailer(.6666);
    }
    h3 {
        @include adjust-font-size-to($base-font-size*1.5, 3);
        @include leader(.5);
        @include trailer(.6666);
    }
    h4 {
        @include adjust-font-size-to($base-font-size*1.25, 2);
        @include leader(.5);
        @include trailer(1.6);
    }
    h5 {
        @include adjust-font-size-to($base-font-size*1.125, 2.5);
        @include leader(.5);
        @include trailer(1.5);
    }
    h6 {
        @include adjust-font-size-to($base-font-size, 2);
        @include leader(0);
        @include trailer(0);
    }
    blockquote {
        background: #f2f2f2;
        font-style: normal;
        display: block;
        @include adjust-font-size-to($base-font-size);
        border-left: 5px solid $primaryColor;
        padding: $base-font-size*2;
        box-sizing: border-box;
        margin-left:.5em;
        margin-right: .5em;
    }
    ul,
    ol {}
    p,
    ul,
    ol,
    dd,
    figure {
        @include adjust-font-size-to($base-font-size, 2); // @include trailer(1);
        @include leader(0);
        @include trailer(2);
    }
    img {
        max-width: 100%;
        @media (max-width: 767px) {
            width: 100%;
            clear: both;
        }
    }
}
